404 static Vector3f slerpUnclamped(Vector3f start, Vector3f end, float delta) {
405  // Dot product - the cosine of the angle between 2 vectors
406  float dot = start.dot(end);
407 
408  // Clamp it to be in the range of Acos()
409  // This may be unnecessary, but floating point
410  // precision can be a fickle mistress.
411  dot = clamp(dot, -1.0f, 1.0f);
412 
413  // Acos(dot) returns the angle between start and end,
414  // And multiplying that by percent returns the angle between
415  // start and the final result.
416  float theta = acosf(dot) * delta;
417  Vector3f relativeVec = (end - start * dot).normalized();
418 
419  // Orthonormal basis
420  // The final result.
421  return ((start * cosf(theta)) + (relativeVec * sinf(theta)));
422 }
